#include<iostream>
#include<string>
using namespace std;

template<class T>
void input(T arr[] , int size){
    cout<<"\nEnter Array Elements .. "<<endl;
    for (int i=0 ; i<size ; i++){
        cout<<"Enter Element "<<i+1<<" : ";
        cin>>arr[i];
    }
}

template<class T>
void display(T arr[] , int size){
    for (int i=0 ; i<size ; i++){
        cout<<arr[i]<<"  ";
    }
    cout<<"\n";
}

template<class T>
void sort(T arr[] , int size){
    for (int i=0 ; i<size-1 ; i++){
        int min_pos = i;
        for (int j=i+1 ; j<size ; j++){
            if(arr[j]<arr[min_pos]){
                min_pos=j;
            }
        }
        swap(arr[i],arr[min_pos]);
    }
}

int main()
{
    int n1,n2;
    cout<<"\nEnter No of Elements to Store in Integer Array - ";
    cin>>n1;
    int arr[n1];
    input(arr,n1);
    cout<<"\nArray is - ";
    display(arr,n1);
    cout<<"\nArray After Sorting - ";
    sort(arr,n1);
    display(arr,n1);

    cout<<"\nEnter No. of Elements to Store in Float Array - ";
    cin>>n2;
    float arr2[n2];
    input(arr2,n2);
    cout<<"\nArray is - ";
    display(arr2,n2);
    cout<<"\nArray After Sorting - ";
    sort(arr2,n2);
    display(arr2,n2);

    return 0;
}
