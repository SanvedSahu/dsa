#include <iostream>
#include <string>

using namespace std;

class Student {
public:
    string name, class_, dob, blood_group, address, license_no;
    int roll_no;
    char division;
    long long phone_no;

    // Constructor
    Student() : roll_no(0), division('\0'), phone_no(0) {}

    // Copy Constructor
    Student(const Student& other) {
        this->name = other.name;
        this->roll_no = other.roll_no;
        this->class_ = other.class_;
        this->division = other.division;
        this->dob = other.dob;
        this->blood_group = other.blood_group;
        this->address = other.address;
        this->phone_no = other.phone_no;
        this->license_no = other.license_no;
        cout << "\nCopy constructor called for student: " << this->name << endl;
    }

    // Destructor
    ~Student() {
        cout << "\nDestructor called for student: " << name << endl;
    }

    // Input function
    void inputData() {
        cout << "Enter Name: ";
        cin>>name;
        cout << "Enter Roll No: ";
        cin >> roll_no;
        cout << "Enter Class: ";
        cin >> class_;
        cout << "Enter Division: ";
        cin >> division;
        cout << "Enter Date of Birth (DD/MM/YYYY): ";
        cin >> dob;
        cout << "Enter Blood Group: ";
        cin >> blood_group;
        cout << "Enter Address: ";
        cin>>address;
        cout << "Enter Phone Number: ";
        cin >> phone_no;
        cout << "Enter License Number: ";
        cin >> license_no;
    }

    // Display function
    inline void displayData() const {
        cout << "Name: " << name << endl
             << "Roll No: " << roll_no << endl
             << "Class: " << class_ << endl
             << "Division: " << division << endl
             << "Date of Birth: " << dob << endl
             << "Blood Group: " << blood_group << endl
             << "Address: " << address << endl
             << "Phone Number: " << phone_no << endl
             << "License Number: " << license_no << endl;
    }
};

int main() {
    int n;
    cout << "Enter the number of students: ";
    cin >> n;

    Student* students = new Student[n];

    for (int i = 0; i < n; ++i) {
        cout << "\nEntering details for Student " << i + 1 << ":\n";
        students[i].inputData();
    }

    cout << "\nStudent Information:\n";
    for (int i = 0; i < n; ++i) {
        cout << "\nDetails of Student " << i + 1 << ":\n";
        students[i].displayData();
    }

    // Demonstrating the copy constructor
    if (n > 0) {
        cout << "\nCreating a copy of the first student:\n";
        Student copiedStudent = students[0]; // Invokes the copy constructor
        cout << "Copied Student Details:\n";
        copiedStudent.displayData();
    }

    // Cleanup
    delete[] students;

    return 0;
}
